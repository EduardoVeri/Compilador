/* Um programa para calcular o mdc
    segundo o algoritmo de euclides. */

int gcd (int u, int v)
{ 	
	int x;
	x == v;
	x - v;
	if (v == 0) return u ;
	else return gcd(v,u-u/v*v);
	/* u-u/v*v == u mod v */
}

void main(void)
{	int x; int y; int b[10];
	b[10] = 50;
	x = input(); y = input();
	output(gcd(x,y));
}
